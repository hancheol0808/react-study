■ router를 사용하는 이유
 ● spw(Single Page Web)을 구현한다
 ● 설치
  npm install react-router-dom
 ● 구조
  <BrowserRouter> //라우터를 사용할 수 있도록 전체 앱을 감싼다
    <CompHeader/>
      <Routes> // 주소에 따라 변경되는 컴퍼넌트를 포함한다
        <Route path="/" element={<CompHome/>} /> // 주소에 home이 출력되는 <CompHome/>컨포넌트를 출력한다
        <Route path="about" element={<CompAbout/>} />
        <Route path="portfoilo" element={<CompPortfoilo/>} />
      </Routes>
    <CompFooter/>
  </BrowserRouter>
  ● 네비게이트 구조
   <Link to="portfoilo">portfoilo</Link>//출력시에는 anchor로 출력됨
